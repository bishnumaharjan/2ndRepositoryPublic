package com.superdemo;public class ChildStudent extends Student {	public String address;	public ChildStudent(int id, String name, String address) {		super(id,name);				this.address = address;	}		public void show() {		System.out.println(id+ " "+ name + " "+address);	}}
package com.superdemo;public class Student {		public int id;	public String name;		public Student(int id, String name) {		this.id = id;		this.name = name;			}		public Student() {			}		public void display() {		System.out.println(id + " "+name);	}}
package com.superdemo;public class DemoSuper {		public static void main(String[] args) {				ChildStudent childStudent = new ChildStudent(100, "subhendu", "TX");				childStudent.show();	}}
